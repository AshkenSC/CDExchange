CREATE TABLE ADMIN (
	ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	USERNAME VARCHAR(255) NOT NULL, 
	PASSWORD VARCHAR(255) NOT NULL,
	STATUS VARCHAR(20) DEFAULT 'active'  NOT NULL
);

CREATE TABLE CUSTOMERS(
    CUSTOMERID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), 
    FIRSTNAME VARCHAR(100) NOT NULL, 
    LASTNAME VARCHAR(100) NOT NULL, 
    EMAIL VARCHAR(255) NOT NULL, PHONE VARCHAR(50) NOT NULL, 
    USERNAME VARCHAR(100) NOT NULL,
    PASSWORD VARCHAR(100) NOT NULL
);


CREATE TABLE ROOM_TYPES (
	TYPE_ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), 
	ROOMTYPE VARCHAR(80) NOT NULL, 
	ROOMDETAIL VARCHAR(150) NOT NULL, 
	PRICE DOUBLE NOT NULL
);

CREATE TABLE ROOM_STATUS (
	ROOM_STATUS_ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	STATUS VARCHAR(50) NOT NULL
	);


CREATE TABLE ROOMS (
	ROOMID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), 
	TYPE_ID INTEGER NOT NULL REFERENCES ROOM_TYPES(TYPE_ID) ON DELETE CASCADE , 
	ROOMNO INTEGER NOT NULL, 
	FLOORNO INTEGER NOT NULL,
	ROOM_STATUS_ID INTEGER NOT NULL REFERENCES ROOM_STATUS(ROOM_STATUS_ID) ON DELETE CASCADE
);





CREATE TABLE RESERVATION (
	RESERVATIONID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), 
		CUSTOMERID INTEGER NOT NULL REFERENCES CUSTOMERS(CUSTOMERID) ON DELETE CASCADE, 
		ROOMID INTEGER NOT NULL REFERENCES ROOMS(ROOMID) ON DELETE CASCADE, 
		STARTDATE DATE NOT NULL, 
		ENDDATE DATE NOT NULL, 
		SPECIALREQUEST VARCHAR(255)
);


CREATE TABLE TRANSACTIONS (
	TRANSACTIONID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), 
	CUSTOMERID INTEGER NOT NULL REFERENCES CUSTOMERS(CUSTOMERID) ON DELETE CASCADE, 
	PAYMENTTYPE VARCHAR(30) DEFAULT 'credit'  NOT NULL, 
	PRICE DOUBLE NOT NULL, CARDHOLDER VARCHAR(255), 
	TRANSACTIONNAME VARCHAR(255), 
	CARDNO VARCHAR(255), EXPIRYDATE VARCHAR(255)
);
